def error_messages
  if object.errors.full_messages.any?
    content_tag(:div, :class => "error_messages") do
      content_tag(:h2, "Invalid Fields") +
      content_tag(:ul) do
        object.errors.full_messages.map do |msg|
          content_tag(:li, msg)
        end.join.html_safe
      end
    end
  end
end
